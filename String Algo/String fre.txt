import java.util.*;
public class Main
{
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int n=sc.nextInt();
		String ar[]=new String[n];
		for(int i=0;i<n;i++){
		    ar[i]=sc.next();
		}
		HashMap<String,Integer> hm=new HashMap<>();
		for(int i=0;i<n;i++){
		    hm.put(ar[i],hm.getOrDefault(ar[i],0)+1);
		}
		for(int i=0;i<n;i++){
		    System.out.println(hm.get(i));
		}
	}
}
